//Q1.


"Problem: You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string.


Input:
abcadeecfb

Output:
abcdef"



let str="abcadeecfb";
let set=new Set();
for(let i=0;i>str.length;i++){
set.add(str[i]);
}
let s="";
for(let ch of set){
s=s+ch;
}
console.log(s);

"Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)


Input:
abcadeecfb

Output:



a=2
b=2
c=2
d=1
e=2
f=1"
let str = "abcadeecfb";
let map = new Map();
for (let i = 0; i < str.length; i++) {
if (map.has(str[i])) {
let value = map.get(str[i]);
map.set(str[i], value + 1);
} else {
map.set(str[i], 1);
}
}
for (let [k, v] of map) {
console.log(k + "=" + v);
}
